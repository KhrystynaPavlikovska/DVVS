#include <iostream> 
#include <stdio.h>
using namespace std;
int main()
{
	int a[7] = { 7,2,2,4,5,6,7 };
	int n = 7; // Number of elements in the array 
	int max = 0;
	int i, k, j = 0;

	for (i = 1; i < n; i++)
	{
		if (max < a[i])
		{
			max = a[i];
		}
	}

	int count = 0;
	int flag = 0;

	for (k = 0; k < n;)
	{
		for (j = 0; j < n; ++j)
		{
			if (a[j] == max)
			{
				if (flag == 0)
				{
					++count;
					flag = 1;
				}
				k++;
			}
		}
		--max;
		flag = 0;
	}
	
	printf("Unique elements : %d\n", count);
	system("pause");
}
